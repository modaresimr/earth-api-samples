var map;

google.load("maps", "2.xx");

function init() {
  map = new GMap2(document.getElementById('map3d'));
  map.addControl(new GMapTypeControl());
  map.setCenter(new GLatLng(37, -122));
  map.addMapType(G_SATELLITE_3D_MAP);
  
  var directions = new GDirections(map);
  map.getEarthInstance(function(o) { o.getWindow().setVisibility(true); } );
  directions.load("from: Haight and Pierce St San Francisco, CA to: Haight and Divisadero San Francisco CA");
}
